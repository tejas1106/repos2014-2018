#include<stdio.h>
#include<conio.h>
#include<time.h>
void main()
{
	int a[20],n,i,j,temp;
	clock_t start,end;
	double time_taken;
	clrscr();
	printf("Enter no. of elements to be sorted : ");
	scanf("%d",&n);
	printf("Enter elements : \n");
	for(i=0;i<n;i++)
	{
		printf("Element %d : ",i+1);
		scanf("%d",&a[i]);
	}

	start=clock();
	//insertion sort
	for(i=1;i<n;i++)
	{
		temp=a[i];
		for(j=i-1;j>=0 && a[j]>temp;j--)
		{
			a[j+1]=a[j];
		}
		a[j+1]=temp;
	}
	end=clock();
	time_taken=((double)end-start)/CLOCKS_PER_SEC;
	printf("Elemetnts after sorting : ");
	for(i=0;i<n;i++)
	{
		printf("%4d",a[i]);
	}
	printf("\nElapsed Time : %f",time_taken);
	getch();
}
