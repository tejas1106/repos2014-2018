void minmax(int a[],int i,int j,int r[]);
int min,max;
void main()
{
	int i,n,a[20],r[2];
	clrscr();
	printf("Enter no. of elements in list : ");
	scanf("%d",&n);
	printf("Enter elements in list : \n");
	for(i=0;i<n;i++)
	{
		printf("Element %d : ",i+1);
		scanf("%d",&a[i]);
	}
	minmax(a,0,n-1,r);
	printf("Minimum value : %d",r[0]);
	printf("\nMaximum value : %d",r[1]);
	getch();
}

void minmax(int a[],int i,int j,int r[])
{
	int mid,min1,max1;
	if(i==j)
	{
		min=max=a[i];
	}
	else
	if(i==j-1)
	{
		if(a[i]<a[j])
		{
			min=a[i];
			max=a[j];
		}
		else
		{
			min=a[j];
			max=a[i];
		}
	}
	else
	{
		mid=(i+j)/2;
		minmax(a,i,mid,r);
			min1=min;
			max1=max;
		minmax(a,mid+1,j,r);
		if(min>min1)
		{
			min=min1;
		}
		if(max<max1)
		{
			max=max1;
		}
		r[0]=min;
		r[1]=max;
	}
}
