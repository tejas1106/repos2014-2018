#include<stdio.h>
#include<conio.h>
void printLCS(char d[][20],int i,int j,char b[]);

void main()
{
	int c[20][20],i,j,m,n;
	char a[20],b[20],d[20][20];
	clrscr();
	printf("Enter parent string : ");
	gets(a+1);
	printf("Enter subsequence : ");
	gets(b+1);
	m=strlen(a)-1;
	printf("%d",m);
	printf("%d",n);
	n=strlen(b)-1;
	for(i=0;i<=m;i++)
	{
		c[0][i]=0;
	}
	for(i=1;i<=n;i++)
	{
		c[i][0]=0;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(b[i]==a[j])
			{
					c[i][j]=c[i-1][j-1]+1;
					d[i][j]='*';
			}
			else
			{
				if(c[i-1][j]>=c[i][j-1])
				{
					c[i][j]=c[i-1][j];
					d[i][j]='#';
				}
				else
				{
					c[i][j]=c[i][j-1];
					d[i][j]=' ';
				}
			}
		}
	}
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=m;j++)
		{
			printf("%4d",c[i][j]);
		}
	}
       //	printf("%d\n",c[m][n]);
	printLCS(d,m,n,b);
	getch();
}

void printLCS(char d[][20],int i,int j,char b[])
{
	if(i==0 || j==0)
	return;
	if(d[i][j]=='*')
	{
		printLCS(d,i-1,j-1,b);
		printf("-->%c",b[i]);
	}
	else
	{
		if(d[i][j]=='#')
		{
			printLCS(d,i-1,j,b);
		}
		else
		printLCS(d,i,j-1,b);
	}
}
