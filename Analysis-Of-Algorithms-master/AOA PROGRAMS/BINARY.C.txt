int binarySearch(int a[],int first,int last,int key)
{
	int mid;
	if(first<last)
	{
		mid=(first+last)/2;
		if(a[mid]<key)
		{
			first=mid+1;
			return binarySearch(a,first,last,key);
		}
		else
		if(a[mid]>key)
		{
			last=mid-1;
			return binarySearch(a,first,last,key);
		}
		else
		return mid;
	}
}
