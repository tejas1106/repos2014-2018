#include<stdio.h>
#include<conio.h>
#include<graphics.h>
void floodFill(int x,int y,int newColor,int defaultColor);
void main()
{
	int graphics_driver=DETECT,graphics_mode;
	int a[20][2],i,n,newColor=11,defaultColor,x,y;
	clrscr();
	initgraph(&graphics_driver,&graphics_mode,"C:\\TC\\BGI");
	printf("Enter no.of vertices of polygon : ");
	scanf("%d",&n);
	printf("Enter vertices of polygon : \n");
	for(i=0;i<n;i++)
	{
		printf("X%d,Y%d : ",i+1,i+1);
		scanf("%d%d",&a[i][0],&a[i][1]);
	}
	a[n][0]=a[0][0];
	a[n][1]=a[0][1];
	printf("Inside point : ");
	scanf("%d%d",&x,&y);
	defaultColor=getpixel(x,y);
	cleardevice();
	setcolor(15);
	for(i=0;i<n;i++)
	{
		line(a[i][0],a[i][1],a[i+1][0],a[i+1][1]);
	}
	x=125;
	y=125;
	floodFill(x,y,newColor,defaultColor);
	getch();
	closegraph();
}

void floodFill(int x,int y,int newColor,int defaultColor)
{
	int color=getpixel(x,y);
	if(color==defaultColor )
	{
		putpixel(x,y,newColor);
		floodFill(x,y+1,newColor,defaultColor);
		floodFill(x,y-1,newColor,defaultColor);
		floodFill(x+1,y,newColor,defaultColor);
		floodFill(x+1,y+1,newColor,defaultColor);
		floodFill(x+1,y-1,newColor,defaultColor);
		floodFill(x-1,y,newColor,defaultColor);
		floodFill(x-1,y+1,newColor,defaultColor);
		floodFill(x-1,y-1,newColor,defaultColor);
	}
}
